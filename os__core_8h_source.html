00819 static inline void os_start(uint32_t ref_hz) {
00820     extern void os_setup_scheduler(uint32_t);
00821 #if CONFIG_OS_SCHEDULER_TYPE != CONFIG_OS_SCHEDULER_COOPERATIVE
00822     /* Setup the scheduler */
00823     os_setup_scheduler(ref_hz);
00824 #endif
00825     /* Launch the scheduler */
00826     os_yield();
00827     /* Idle loop. This loop will control the application process but also
00828      * the event process.
00829      * They both run inside an infinite loop. They cannot run at the same
00830      * time.
00831      */
00832     while (true) {
00833         /* If the event process is running, call the event scheduler */
00834         if (__os_process_is_event(__os_process_get_current())) {
00835             __os_event_scheduler();
00836         }
00837         /* Else it means the application process is running. In other
00838          * word, not other processes are actives or pending for events.
00839          */
00840         else {
00841             HOOK_OS_IDLE();
00842         }
00843     }
00844 }
