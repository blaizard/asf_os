00292 #if UC3C
00293 static long int pcl_configure_clocks_uc3c(pcl_freq_param_t *param)
00294 {
00295   #define PM_MAX_MUL                         ((1 << AVR32_SCIF_PLLMUL_SIZE) - 1)
00296   #define AVR32_PM_PBA_MAX_FREQ              66000000
00297   #define AVR32_PM_PLL_VCO_RANGE0_MAX_FREQ   240000000
00298   #define AVR32_PM_PLL_VCO_RANGE0_MIN_FREQ   160000000
00299 
00300     // Implementation for  UC3C parts.
00301         // Supported frequencies:
00302         // Fosc0 mul div PLL div2_en cpu_f pba_f   Comment
00303         //  12   15   1  192     1     12    12
00304         //  12    9   3   40     1     20    20    PLL out of spec
00305         //  12   15   1  192     1     24    12
00306         //  12    9   1  120     1     30    15
00307         //  12    9   3   40     0     40    20    PLL out of spec
00308         //  12   15   1  192     1     48    12
00309         //  12   15   1  192     1     48    24
00310         //  12    8   1  108     1     54    27
00311         //  12    9   1  120     1     60    15
00312         //  12    9   1  120     1     60    30
00313         //  12   10   1  132     1     66    16.5
00314         //
00315         unsigned long in_cpu_f  = param->cpu_f;
00316         unsigned long in_osc0_f = param->osc0_f;
00317         unsigned long mul, div, div2_en = 0, div2_cpu = 0, div2_pba = 0;
00318         unsigned long pll_freq, rest;
00319         bool b_div2_pba, b_div2_cpu;
00320 
00321         // Configure OSC0 in crystal mode, external crystal with a FOSC0 Hz frequency.
00322         scif_configure_osc_crystalmode(SCIF_OSC0, in_osc0_f);
00323         // Enable the OSC0
00324         scif_enable_osc(SCIF_OSC0, param->osc0_startup, true);
00325         // Set the main clock source as being OSC0.
00326         pm_set_mclk_source(PM_CLK_SRC_OSC0);
00327 
00328         // Start with CPU freq config
00329         if (in_cpu_f == in_osc0_f)
00330         {
00331           param->cpu_f = in_osc0_f;
00332           param->pba_f = in_osc0_f;
00333           return PASS;
00334         }
00335         else if (in_cpu_f < in_osc0_f)
00336         {
00337           // TBD
00338         }
00339 
00340         rest = in_cpu_f % in_osc0_f;
00341 
00342         for (div = 1; div < 32; div++)
00343         {
00344           if ((div * rest) % in_osc0_f == 0)
00345             break;
00346         }
00347         if (div == 32)
00348           return FAIL;
00349 
00350         mul = (in_cpu_f * div) / in_osc0_f;
00351 
00352         if (mul > PM_MAX_MUL)
00353           return FAIL;
00354 
00355         // export 2power from PLL div to div2_cpu
00356         while (!(div % 2))
00357         {
00358           div /= 2;
00359           div2_cpu++;
00360         }
00361 
00362         // Here we know the mul and div parameter of the PLL config.
00363         // . Check out if the PLL has a valid in_cpu_f.
00364         // . Try to have for the PLL frequency (VCO output) the highest possible value
00365         //   to reduce jitter.
00366         while (in_osc0_f * 2 * mul / div < AVR32_PM_PLL_VCO_RANGE0_MAX_FREQ)
00367         {
00368           if (2 * mul > PM_MAX_MUL)
00369             break;
00370           mul *= 2;
00371           div2_cpu++;
00372         }
00373 
00374         if (div2_cpu != 0)
00375         {
00376           div2_cpu--;
00377           div2_en = 1;
00378         }
00379 
00380         pll_freq = in_osc0_f * mul / (div * (1 << div2_en));
00381 
00382         // Update real CPU Frequency
00383         param->cpu_f = pll_freq / (1 << div2_cpu);
00384         mul--;
00385 
00386         scif_pll_opt_t opt;
00387 
00388         opt.osc = SCIF_OSC0,     // Sel Osc0 or Osc1
00389         opt.lockcount = 16,      // lockcount in main clock for the PLL wait lock
00390         opt.div = div,             // DIV=1 in the formula
00391         opt.mul = mul,             // MUL=7 in the formula
00392         opt.pll_div2 = div2_en,        // pll_div2 Divide the PLL output frequency by 2 (this settings does not change the FVCO value)
00393         opt.pll_wbwdisable = 0,  //pll_wbwdisable 1 Disable the Wide-Bandith Mode (Wide-Bandwith mode allow a faster startup time and out-of-lock time). 0 to enable the Wide-Bandith Mode.
00394         opt.pll_freq = (pll_freq < AVR32_PM_PLL_VCO_RANGE0_MIN_FREQ) ? 1 : 0,        // Set to 1 for VCO frequency range 80-180MHz, set to 0 for VCO frequency range 160-240Mhz.
00395 
00396 
00397         scif_pll_setup(SCIF_PLL0, opt); // lockcount in main clock for the PLL wait lock
00398 
00399         /* Enable PLL0 */
00400         scif_pll_enable(SCIF_PLL0);
00401 
00402         /* Wait for PLL0 locked */
00403         scif_wait_for_pll_locked(SCIF_PLL0) ;
00404 
00405         rest = pll_freq;
00406         while (rest > AVR32_PM_PBA_MAX_FREQ ||
00407                rest != param->pba_f)
00408         {
00409           div2_pba++;
00410           rest = pll_freq / (1 << div2_pba);
00411           if (rest < param->pba_f)
00412             break;
00413         }
00414 
00415         // Update real PBA Frequency
00416         param->pba_f = pll_freq / (1 << div2_pba);
00417 
00418 
00419         if (div2_cpu)
00420         {
00421           b_div2_cpu = true;
00422           div2_cpu--;
00423         }
00424         else
00425           b_div2_cpu = false;
00426 
00427         if (div2_pba)
00428         {
00429           b_div2_pba = true;
00430           div2_pba--;
00431         }
00432         else
00433           b_div2_pba = false;
00434 
00435         if (b_div2_cpu == true )
00436         {
00437           pm_set_clk_domain_div(PM_CLK_DOMAIN_0, (pm_divratio_t) div2_cpu); // CPU
00438           pm_set_clk_domain_div(PM_CLK_DOMAIN_1, (pm_divratio_t) div2_cpu); // HSB
00439           pm_set_clk_domain_div(PM_CLK_DOMAIN_3, (pm_divratio_t) div2_cpu); // PBB
00440         }
00441         if (b_div2_pba == true )
00442         {
00443           pm_set_clk_domain_div(PM_CLK_DOMAIN_2, (pm_divratio_t) div2_pba); // PBA
00444           pm_set_clk_domain_div(PM_CLK_DOMAIN_4, (pm_divratio_t) div2_pba); // PBC
00445         }
00446 
00447         // Set Flashc Wait State
00448         flashc_set_flash_waitstate_and_readmode(param->cpu_f);
00449 
00450         // Set the main clock source as being PLL0.
00451         pm_set_mclk_source(PM_CLK_SRC_PLL0);
00452 
00453         return PASS;
00454 }
