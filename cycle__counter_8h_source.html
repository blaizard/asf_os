00188 static inline unsigned long cpu_is_timeout(t_cpu_time *cpu_time)
00189 {
00190   unsigned long current_cycle_count = Get_system_register(AVR32_COUNT);
00191 
00192   if( cpu_time->timer_state==CPU_TIMER_STATE_STOPPED )
00193     return false;
00194 
00195   // Test if the timeout as already occured.
00196   else if (cpu_time->timer_state == CPU_TIMER_STATE_REACHED)
00197     return true;
00198 
00199   // If the ending cycle count of this timeout is wrapped, ...
00200   else if (cpu_time->delay_start_cycle > cpu_time->delay_end_cycle)
00201   {
00202     if (current_cycle_count < cpu_time->delay_start_cycle && current_cycle_count > cpu_time->delay_end_cycle)
00203     {
00204       cpu_time->timer_state = CPU_TIMER_STATE_REACHED;
00205       return true;
00206     }
00207     return false;
00208   }
00209   else
00210   {
00211     if (current_cycle_count < cpu_time->delay_start_cycle || current_cycle_count > cpu_time->delay_end_cycle)
00212     {
00213       cpu_time->timer_state = CPU_TIMER_STATE_REACHED;
00214       return true;
00215     }
00216     return false;
00217   }
00218 }
