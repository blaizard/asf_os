{
    enum os_event_status status = OS_EVENT_NONE;
    struct os_semaphore *sem = (struct os_semaphore *) args;

    os_enter_critical();
    if (sem->counter == 1) {
        sem->counter = 0;
        status = OS_EVENT_OK_STOP;
    }
    else if (sem->counter > 1) {
        sem->counter--;
        status = OS_EVENT_OK_CONTINUE;
    }
    os_leave_critical();

    return status;
}
